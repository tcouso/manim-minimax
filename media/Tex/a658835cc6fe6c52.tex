                                 utility(s) \qquad \text{si $s$ es terminal} \cr
                                 max\{minimax(\text{$s'$}| $s'$ \text{es hijo de }$s$)\}\qquad \text{si es el turno de Max en $s$} \cr
                                 min\{minimax(\text{$s'$}| $s'$ \text{es hijo de }$s$)\})\}\qquad \text{si es el turno de Min en $s$} \cr
                                 \end{cases}
\end{align*}
